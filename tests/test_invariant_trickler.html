#include #include #include #include #include #include #include #include #include /* Configuration - adjust host/port to match the device under test */ #ifndef TARGET_HOST #define TARGET_HOST "127.0.0.1" #endif #ifndef TARGET_PORT #define TARGET_PORT 80 #endif #define CONNECT_TIMEOUT_SEC 2 #define RECV_TIMEOUT_SEC 3 typedef struct { const char *description; const char *auth_header; /* NULL means no Authorization header at all */ const char *cookie_header; /* NULL means no Cookie header at all */ } auth_scenario_t; /* * Send a minimal HTTP GET request to the given path with optional * auth/cookie headers. Returns the HTTP status code, or -1 on error. */ static int send_http_request(const char *path, const char *auth_header, const char *cookie_header) { int sock; struct sockaddr_in addr; char request[2048]; char response[4096]; int pos = 0; int status_code = -1; ssize_t bytes; struct timeval tv; sock = socket(AF_INET, SOCK_STREAM, 0); if (sock < 0) { return -1; } /* Set receive timeout */ tv.tv_sec = RECV_TIMEOUT_SEC; tv.tv_usec = 0; setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv)); /* Set send timeout */ tv.tv_sec = CONNECT_TIMEOUT_SEC; tv.tv_usec = 0; setsockopt(sock, SOL_SOCKET, SO_SNDTIMEO, &tv, sizeof(tv)); memset(&addr, 0, sizeof(addr)); addr.sin_family = AF_INET; addr.sin_port = htons(TARGET_PORT); addr.sin_addr.s_addr = inet_addr(TARGET_HOST); if (connect(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) { close(sock); return -1; } /* Build request */ pos += snprintf(request + pos, sizeof(request) - pos, "GET %s HTTP/1.1\r\n" "Host: %s:%d\r\n" "Connection: close\r\n", path, TARGET_HOST, TARGET_PORT); if (auth_header != NULL) { pos += snprintf(request + pos, sizeof(request) - pos, "Authorization: %s\r\n", auth_header); } if (cookie_header != NULL) { pos += snprintf(request + pos, sizeof(request) - pos, "Cookie: %s\r\n", cookie_header); } pos += snprintf(request + pos, sizeof(request) - pos, "\r\n"); if (send(sock, request, pos, 0) < 0) { close(sock); return -1; } /* Read response */ memset(response, 0, sizeof(response)); bytes = recv(sock, response, sizeof(response) - 1, 0); close(sock); if (bytes <= 0) { return -1; } /* Parse status line: "HTTP/1.x NNN ..." */ if (strncmp(response, "HTTP/", 5) == 0) { char *space = strchr(response, ' '); if (space != NULL) { status_code = atoi(space + 1); } } return status_code; } /* * Helper: assert that a status code represents an authentication/ * authorisation rejection.
